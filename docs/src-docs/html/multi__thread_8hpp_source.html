32  class SharedQueue

34  public:
35  T pop() {
36  std::unique_lock<std::mutex> mlock(mutex_);
37 
38  while (queue_.empty()) {
39  cond_.wait(mlock);
40  }
41 
42  auto val = queue_.front();
43 
44  queue_.pop();
45 
46  return val;
47  }

49  void pop(T& item) {
50  std::unique_lock<std::mutex> mlock(mutex_);
51 
52  while (queue_.empty()) {
53  cond_.wait(mlock);
54  }
55 
56  item = queue_.front();
57  queue_.pop();
58  }

60  void push(const T& item) {
61  std::unique_lock<std::mutex> mlock(mutex_);
62  queue_.push(item);
63  mlock.unlock();
64  cond_.notify_one();
65  }

78  class Parallel
79  {
80 
81  public:
82  void Start(unsigned n_threads) {
83  // Figure out how many threads we can actually use.
84  unsigned n_used_threads = n_threads;
85 
86  // Call the initialize routine for the subclass.
87  Init(n_threads);
88 
89  // list of threads
90  std::vector<std::thread> threads(n_threads);
91  // pointer to a shred queue
92  SharedQueue<std::shared_ptr<T>>* q_ptr = new SharedQueue<std::shared_ptr<T>>;
93  // Start each thread.
94  // the thread's queue, and start each thread.
95  for (unsigned i = 0; i < n_threads; ++i)
96  {
97  threads[i] = std::thread(&Parallel::Thread, this, i, q_ptr);
98  }
99 
100  std::shared_ptr<T> item_ptr = nullptr;
101 
102  // Get the items from the subclass.
103  // For each item get the item size and add it the queue of the
104  // thread with the least amount of load.
105  for (std::shared_ptr<T> item_ptr = NextItem(); item_ptr != nullptr; item_ptr = NextItem())
106  {
107  q_ptr->push(item_ptr);
108  }
109 
110  // Join all the threads.
111  // Pass a null pointer to tell the threads not to expect
112  // anymore items.
113  // Clean up the queue memory.
114  for (unsigned i = 0; i < n_threads; ++i) {
115  q_ptr->push(nullptr);
116  }
117 
118  for (unsigned i = 0; i < n_threads; ++i) {
119  threads[i].join();
120  }
121 
122  delete q_ptr;
123 
124  // Call the subclass close method.
125  Close();
126  }
