25 template <class Mutex>
26 class ansicolor_sink: public base_sink<Mutex>
27 {
28 public:
29  ansicolor_sink(FILE* file): target_file_(file)
30  {
31  should_do_colors_ = details::os::in_terminal(file) && details::os::is_color_terminal();
32  colors_[level::trace] = "\033[36m"; // cyan;
33  colors_[level::debug] = "\033[36m"; // cyan;
34  colors_[level::info] = "\033[0m";// default color;
35  colors_[level::warn] = "\033[33m\033[1m"; // yellow_bold;
36  colors_[level::err] = "\033[31m\033[1m"; // red_bold;
37  colors_[level::critical] = "\033[1m\033[41m"; // bold_red_bg;
38  colors_[level::off] = "\033[0m"; //reset to default;
39  }
40  virtual ~ansicolor_sink()
41  {
42  _flush();
43  }
44 
45 
46 
47 protected:
48  virtual void _sink_it(const details::log_msg& msg) override
49  {
50  // Wrap the originally formatted message in color codes.
51  // If color is not supported in the terminal, log as is instead.
52  if (should_do_colors_)
53  {
54  const std::string& prefix = colors_[msg.level];
55  const std::string& reset = colors_[level::off];
56  fwrite(prefix.data(), sizeof(char), prefix.size(), target_file_);
57  fwrite(msg.formatted.data(), sizeof(char), msg.formatted.size(), target_file_);
58  fwrite(reset.data(), sizeof(char), reset.size(), target_file_);
59  }
60  else
61  {
62  fwrite(msg.formatted.data(), sizeof(char), msg.formatted.size(), target_file_);
63  }
64  _flush();
65  }
66 
67  void _flush() override
68  {
69  fflush(target_file_);
70  }
71  FILE* target_file_;
72  bool should_do_colors_;
73  std::map<level::level_enum, std::string> colors_;
74 };
